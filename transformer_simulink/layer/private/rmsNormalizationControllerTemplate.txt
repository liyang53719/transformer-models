function [gBeat, gBeatValid, captureInvRms, sramWriteAddr, sramWriteBank, sramWriteValid, sramReadAddr, sramReadBank, sramReadValid, ddrReadAddr, ddrReadEn, done, busy] = Controller(start, cfgGammaBeat, cfgGammaValid, ddrDataValid, invRmsAccepted)
%#codegen

beatsPerToken = uint16(192);
numTokens = uint8(64);
warmupFlushCycles = uint8(166);

persistent state gammaMem gammaWriteBeat gammaLoaded
persistent loadActive loadTokenIndex nextLoadTokenIndex loadBank
persistent requestBeatIndex receiveBeatIndex sumFlushCount
persistent pendingInvRms pendingTokenIndex pendingBank
persistent readyTokenValid readyTokenIndex readyBank
persistent outputActive outputTokenIndex outputBank outputBeatIndex

if isempty(state)
    state = uint8(0);
    gammaMem = zeros(1, 1536, 'single');
    gammaWriteBeat = uint16(0);
    gammaLoaded = false;
    loadActive = false;
    loadTokenIndex = uint8(0);
    nextLoadTokenIndex = uint8(0);
    loadBank = false;
    requestBeatIndex = uint16(0);
    receiveBeatIndex = uint16(0);
    sumFlushCount = uint8(0);
    pendingInvRms = false;
    pendingTokenIndex = uint8(0);
    pendingBank = false;
    readyTokenValid = false;
    readyTokenIndex = uint8(0);
    readyBank = false;
    outputActive = false;
    outputTokenIndex = uint8(0);
    outputBank = false;
    outputBeatIndex = uint16(0);
end

gBeat = zeros(1, 8, 'single');
gBeatValid = false;
captureInvRms = false;
sramWriteAddr = uint16(0);
sramWriteBank = false;
sramWriteValid = false;
sramReadAddr = uint16(0);
sramReadBank = false;
sramReadValid = false;
ddrReadAddr = uint16(0);
ddrReadEn = false;
done = false;
busy = state ~= uint8(0);

if cfgGammaValid ~= 0
    gammaMem = iWriteBeat(gammaMem, cfgGammaBeat, gammaWriteBeat);
    if gammaWriteBeat == beatsPerToken - 1
        gammaWriteBeat = uint16(0);
        gammaLoaded = true;
    else
        gammaWriteBeat = gammaWriteBeat + uint16(1);
    end
end

switch state
    case uint8(0)
        if start ~= 0 && gammaLoaded
            state = uint8(1);
            loadActive = true;
            loadTokenIndex = uint8(0);
            nextLoadTokenIndex = uint8(1);
            loadBank = false;
            requestBeatIndex = uint16(0);
            receiveBeatIndex = uint16(0);
            sumFlushCount = uint8(0);
            pendingInvRms = false;
            readyTokenValid = false;
            outputActive = false;
            outputBeatIndex = uint16(0);
            busy = true;
        end

    otherwise
        busy = true;

        if pendingInvRms && sumFlushCount > uint8(0)
            if sumFlushCount == uint8(1)
                captureInvRms = true;
            end
            sumFlushCount = sumFlushCount - uint8(1);
        end

        if invRmsAccepted ~= 0 && pendingInvRms
            readyTokenValid = true;
            readyTokenIndex = pendingTokenIndex;
            readyBank = pendingBank;
            pendingInvRms = false;
        end

        if ~outputActive && readyTokenValid
            outputActive = true;
            outputTokenIndex = readyTokenIndex;
            outputBank = readyBank;
            outputBeatIndex = uint16(0);
            readyTokenValid = false;
        end

        if loadActive
            [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, loadTokenIndex, requestBeatIndex, beatsPerToken);
            if ddrDataValid ~= 0
                sramWriteAddr = receiveBeatIndex;
                sramWriteBank = loadBank;
                sramWriteValid = true;

                if receiveBeatIndex == beatsPerToken - 1
                    if warmupFlushCycles == uint8(0)
                        captureInvRms = true;
                    end
                    sumFlushCount = warmupFlushCycles;
                    pendingInvRms = true;
                    pendingTokenIndex = loadTokenIndex;
                    pendingBank = loadBank;
                    loadActive = false;
                    requestBeatIndex = uint16(0);
                    receiveBeatIndex = uint16(0);
                else
                    receiveBeatIndex = receiveBeatIndex + uint16(1);
                end
            end
        end

        if outputActive
            sramReadAddr = outputBeatIndex;
            sramReadBank = outputBank;
            sramReadValid = true;
            gBeat = iReadBeat(gammaMem, outputBeatIndex);
            gBeatValid = true;

            if outputBeatIndex == beatsPerToken - 1
                if outputTokenIndex == numTokens - uint8(1)
                    done = true;
                    state = uint8(0);
                    loadActive = false;
                    pendingInvRms = false;
                    readyTokenValid = false;
                    outputActive = false;
                    nextLoadTokenIndex = uint8(0);
                    requestBeatIndex = uint16(0);
                    receiveBeatIndex = uint16(0);
                    outputBeatIndex = uint16(0);
                    sumFlushCount = uint8(0);
                    busy = false;
                else
                    outputActive = false;
                    outputBeatIndex = uint16(0);
                end
            else
                outputBeatIndex = outputBeatIndex + uint16(1);
            end
        end

        if state ~= uint8(0) && ~loadActive && ~pendingInvRms && nextLoadTokenIndex < numTokens
            [canStartLoad, freeBank] = iGetFreeBank(outputActive, outputBank, pendingInvRms, pendingBank, readyTokenValid, readyBank);
            if canStartLoad
                loadActive = true;
                loadTokenIndex = nextLoadTokenIndex;
                nextLoadTokenIndex = nextLoadTokenIndex + uint8(1);
                loadBank = freeBank;
                requestBeatIndex = uint16(0);
                receiveBeatIndex = uint16(0);
            end
        end
end

end

function mem = iWriteBeat(mem, beat, beatIndex)
baseIndex = uint16(bitshift(beatIndex, 3));
for laneIndex = uint16(0):uint16(7)
    mem(baseIndex + laneIndex + uint16(1)) = beat(laneIndex + uint16(1));
end
end

function beat = iReadBeat(mem, beatIndex)
beat = zeros(1, 8, 'single');
baseIndex = uint16(bitshift(beatIndex, 3));
for laneIndex = uint16(0):uint16(7)
    beat(laneIndex + uint16(1)) = mem(baseIndex + laneIndex + uint16(1));
end
end

function [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, tokenIndex, requestBeatIndex, beatsPerToken)
if requestBeatIndex < beatsPerToken
    ddrReadEn = true;
    ddrReadAddr = uint16(uint16(tokenIndex) * beatsPerToken + requestBeatIndex);
    requestBeatIndex = requestBeatIndex + uint16(1);
end
end

function [canStartLoad, freeBank] = iGetFreeBank(outputActive, outputBank, pendingInvRms, pendingBank, readyTokenValid, readyBank)
bank0Occupied = false;
bank1Occupied = false;

if outputActive
    if outputBank
        bank1Occupied = true;
    else
        bank0Occupied = true;
    end
end

if pendingInvRms
    if pendingBank
        bank1Occupied = true;
    else
        bank0Occupied = true;
    end
end

if readyTokenValid
    if readyBank
        bank1Occupied = true;
    else
        bank0Occupied = true;
    end
end

if ~bank0Occupied
    canStartLoad = true;
    freeBank = false;
elseif ~bank1Occupied
    canStartLoad = true;
    freeBank = true;
else
    canStartLoad = false;
    freeBank = false;
end
end
