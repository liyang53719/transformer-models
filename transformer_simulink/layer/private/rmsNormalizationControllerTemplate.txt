function [xBeat, xBeatValid, gBeat, gBeatValid, captureInvRms, ddrReadAddr, ddrReadEn, done, busy] = Controller(reset, start, cfgGammaBeat, cfgGammaValid, ddrDataBeat, ddrDataValid)
%#codegen

beatsPerToken = uint16(192);
numTokens = uint8(64);
warmupFlushCycles = uint8(22);

persistent state gammaMem tokenMemA tokenMemB
persistent gammaWriteBeat gammaLoaded
persistent outputTokenIndex prefetchTokenIndex
persistent requestBeatIndex receiveBeatIndex outputBeatIndex
persistent sumFlushCount loadingActive flushPending nextTokenReady
persistent outputBank fillBank

if isempty(state)
    state = uint8(0);
    gammaMem = zeros(1, 1536, 'single');
    tokenMemA = zeros(1, 1536, 'single');
    tokenMemB = zeros(1, 1536, 'single');
    gammaWriteBeat = uint16(0);
    gammaLoaded = false;
    outputTokenIndex = uint8(0);
    prefetchTokenIndex = uint8(0);
    requestBeatIndex = uint16(0);
    receiveBeatIndex = uint16(0);
    outputBeatIndex = uint16(0);
    sumFlushCount = uint8(0);
    loadingActive = false;
    flushPending = false;
    nextTokenReady = false;
    outputBank = uint8(1);
    fillBank = uint8(2);
end

xBeat = zeros(1, 8, 'single');
xBeatValid = false;
gBeat = zeros(1, 8, 'single');
gBeatValid = false;
captureInvRms = false;
ddrReadAddr = uint16(0);
ddrReadEn = false;
done = false;
busy = state ~= uint8(0);

if reset ~= 0
    state = uint8(0);
    gammaMem = zeros(1, 1536, 'single');
    tokenMemA = zeros(1, 1536, 'single');
    tokenMemB = zeros(1, 1536, 'single');
    gammaWriteBeat = uint16(0);
    gammaLoaded = false;
    outputTokenIndex = uint8(0);
    prefetchTokenIndex = uint8(0);
    requestBeatIndex = uint16(0);
    receiveBeatIndex = uint16(0);
    outputBeatIndex = uint16(0);
    sumFlushCount = uint8(0);
    loadingActive = false;
    flushPending = false;
    nextTokenReady = false;
    outputBank = uint8(1);
    fillBank = uint8(2);
else
    if cfgGammaValid ~= 0
        gammaMem = iWriteBeat(gammaMem, cfgGammaBeat, gammaWriteBeat);
        if gammaWriteBeat == beatsPerToken - 1
            gammaWriteBeat = uint16(0);
            gammaLoaded = true;
        else
            gammaWriteBeat = gammaWriteBeat + uint16(1);
        end
    end

    switch state
        case uint8(0)
            if start ~= 0 && gammaLoaded
                state = uint8(1);
                outputTokenIndex = uint8(0);
                prefetchTokenIndex = uint8(0);
                requestBeatIndex = uint16(0);
                receiveBeatIndex = uint16(0);
                outputBeatIndex = uint16(0);
                sumFlushCount = uint8(0);
                loadingActive = true;
                flushPending = false;
                nextTokenReady = false;
                outputBank = uint8(1);
                fillBank = uint8(2);
                busy = true;
            end

        case uint8(1)
            busy = true;
            [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, outputTokenIndex, requestBeatIndex, beatsPerToken);
            if ddrDataValid ~= 0
                [tokenMemA, tokenMemB] = iWriteBufferedBeat(tokenMemA, tokenMemB, outputBank, receiveBeatIndex, ddrDataBeat);
                if receiveBeatIndex == beatsPerToken - 1
                    receiveBeatIndex = uint16(0);
                    requestBeatIndex = uint16(0);
                    sumFlushCount = warmupFlushCycles;
                    loadingActive = false;
                    flushPending = true;
                    state = uint8(2);
                else
                    receiveBeatIndex = receiveBeatIndex + uint16(1);
                end
            end

        case uint8(2)
            busy = true;
            if sumFlushCount == uint8(0)
                captureInvRms = true;
                flushPending = false;
                nextTokenReady = false;
                outputBeatIndex = uint16(0);
                if numTokens > uint8(1)
                    prefetchTokenIndex = uint8(1);
                    requestBeatIndex = uint16(0);
                    receiveBeatIndex = uint16(0);
                    loadingActive = true;
                end
                state = uint8(3);
            else
                sumFlushCount = sumFlushCount - uint8(1);
            end

        case uint8(3)
            busy = true;
            xBeatValid = true;
            gBeatValid = true;
            xBeat = iReadBufferedBeat(tokenMemA, tokenMemB, outputBank, outputBeatIndex);
            gBeat = iReadBeat(gammaMem, outputBeatIndex);

            [tokenMemA, tokenMemB, captureInvRms, ddrReadEn, ddrReadAddr, requestBeatIndex, receiveBeatIndex, sumFlushCount, loadingActive, flushPending, nextTokenReady] = ...
                iAdvanceBackground(tokenMemA, tokenMemB, fillBank, prefetchTokenIndex, ddrDataBeat, ddrDataValid, captureInvRms, ddrReadEn, ddrReadAddr, requestBeatIndex, receiveBeatIndex, sumFlushCount, loadingActive, flushPending, nextTokenReady, beatsPerToken);

            if outputBeatIndex == beatsPerToken - 1
                if outputTokenIndex == numTokens - uint8(1)
                    done = true;
                    state = uint8(0);
                    outputTokenIndex = uint8(0);
                    prefetchTokenIndex = uint8(0);
                    requestBeatIndex = uint16(0);
                    receiveBeatIndex = uint16(0);
                    outputBeatIndex = uint16(0);
                    sumFlushCount = uint8(0);
                    loadingActive = false;
                    flushPending = false;
                    nextTokenReady = false;
                    busy = false;
                elseif nextTokenReady
                    [outputTokenIndex, prefetchTokenIndex, requestBeatIndex, receiveBeatIndex, outputBeatIndex, loadingActive, flushPending, nextTokenReady, outputBank, fillBank] = ...
                        iRotateTokens(outputTokenIndex, prefetchTokenIndex, requestBeatIndex, receiveBeatIndex, outputBeatIndex, loadingActive, flushPending, nextTokenReady, outputBank, fillBank, numTokens);
                else
                    outputBeatIndex = uint16(0);
                    state = uint8(4);
                end
            else
                outputBeatIndex = outputBeatIndex + uint16(1);
            end

        otherwise
            busy = true;
            [tokenMemA, tokenMemB, captureInvRms, ddrReadEn, ddrReadAddr, requestBeatIndex, receiveBeatIndex, sumFlushCount, loadingActive, flushPending, nextTokenReady] = ...
                iAdvanceBackground(tokenMemA, tokenMemB, fillBank, prefetchTokenIndex, ddrDataBeat, ddrDataValid, captureInvRms, ddrReadEn, ddrReadAddr, requestBeatIndex, receiveBeatIndex, sumFlushCount, loadingActive, flushPending, nextTokenReady, beatsPerToken);

            if nextTokenReady
                [outputTokenIndex, prefetchTokenIndex, requestBeatIndex, receiveBeatIndex, outputBeatIndex, loadingActive, flushPending, nextTokenReady, outputBank, fillBank] = ...
                    iRotateTokens(outputTokenIndex, prefetchTokenIndex, requestBeatIndex, receiveBeatIndex, outputBeatIndex, loadingActive, flushPending, nextTokenReady, outputBank, fillBank, numTokens);
                state = uint8(3);
            end
    end
end

end

function mem = iWriteBeat(mem, beat, beatIndex)
baseIndex = uint16(bitshift(beatIndex, 3));
for laneIndex = uint16(0):uint16(7)
    mem(baseIndex + laneIndex + uint16(1)) = beat(laneIndex + uint16(1));
end
end

function beat = iReadBeat(mem, beatIndex)
beat = zeros(1, 8, 'single');
baseIndex = uint16(bitshift(beatIndex, 3));
for laneIndex = uint16(0):uint16(7)
    beat(laneIndex + uint16(1)) = mem(baseIndex + laneIndex + uint16(1));
end
end

function [tokenMemA, tokenMemB] = iWriteBufferedBeat(tokenMemA, tokenMemB, bankIndex, beatIndex, beat)
if bankIndex == uint8(1)
    tokenMemA = iWriteBeat(tokenMemA, beat, beatIndex);
else
    tokenMemB = iWriteBeat(tokenMemB, beat, beatIndex);
end
end

function beat = iReadBufferedBeat(tokenMemA, tokenMemB, bankIndex, beatIndex)
if bankIndex == uint8(1)
    beat = iReadBeat(tokenMemA, beatIndex);
else
    beat = iReadBeat(tokenMemB, beatIndex);
end
end

function [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, tokenIndex, requestBeatIndex, beatsPerToken)
if requestBeatIndex < beatsPerToken
    ddrReadEn = true;
    ddrReadAddr = uint16(uint16(tokenIndex) * beatsPerToken + requestBeatIndex);
    requestBeatIndex = requestBeatIndex + uint16(1);
end
end

function [tokenMemA, tokenMemB, captureInvRms, ddrReadEn, ddrReadAddr, requestBeatIndex, receiveBeatIndex, sumFlushCount, loadingActive, flushPending, nextTokenReady] = ...
    iAdvanceBackground(tokenMemA, tokenMemB, fillBank, prefetchTokenIndex, ddrDataBeat, ddrDataValid, captureInvRms, ddrReadEn, ddrReadAddr, requestBeatIndex, receiveBeatIndex, sumFlushCount, loadingActive, flushPending, nextTokenReady, beatsPerToken)

startedFlushThisCycle = false;

if loadingActive
    [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, prefetchTokenIndex, requestBeatIndex, beatsPerToken);
end

if ddrDataValid ~= 0
    [tokenMemA, tokenMemB] = iWriteBufferedBeat(tokenMemA, tokenMemB, fillBank, receiveBeatIndex, ddrDataBeat);
    if receiveBeatIndex == beatsPerToken - 1
        receiveBeatIndex = uint16(0);
        requestBeatIndex = uint16(0);
        sumFlushCount = uint8(22);
        loadingActive = false;
        flushPending = true;
        startedFlushThisCycle = true;
    else
        receiveBeatIndex = receiveBeatIndex + uint16(1);
    end
end

if flushPending && ~startedFlushThisCycle
    if sumFlushCount == uint8(0)
        captureInvRms = true;
        flushPending = false;
        nextTokenReady = true;
    else
        sumFlushCount = sumFlushCount - uint8(1);
    end
end
end

function [outputTokenIndex, prefetchTokenIndex, requestBeatIndex, receiveBeatIndex, outputBeatIndex, loadingActive, flushPending, nextTokenReady, outputBank, fillBank] = ...
    iRotateTokens(outputTokenIndex, prefetchTokenIndex, requestBeatIndex, receiveBeatIndex, outputBeatIndex, loadingActive, flushPending, nextTokenReady, outputBank, fillBank, numTokens)

oldOutputBank = outputBank;
outputBank = fillBank;
fillBank = oldOutputBank;
outputTokenIndex = outputTokenIndex + uint8(1);
outputBeatIndex = uint16(0);
requestBeatIndex = uint16(0);
receiveBeatIndex = uint16(0);
flushPending = false;
nextTokenReady = false;

if outputTokenIndex < numTokens - uint8(1)
    prefetchTokenIndex = outputTokenIndex + uint8(1);
    loadingActive = true;
else
    prefetchTokenIndex = outputTokenIndex;
    loadingActive = false;
end
end
