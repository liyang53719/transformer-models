function [gBeat, gBeatValid, captureInvRms, captureInvRmsBank, sramWriteAddr, sramWriteBank, sramWriteValid, sramReadAddr, sramReadBank, sramReadValid, ddrReadAddr, ddrReadEn, done, busy] = Controller(start, cfgGammaBeat, cfgGammaValid, ddrDataValid, invRmsAcceptedBank0, invRmsAcceptedBank1)
%#codegen

beatsPerToken = uint16(192);
numTokens = uint8(64);
warmupFlushCycles = uint8(166);

persistent state gammaMem gammaWriteBeat gammaLoaded
persistent loadActive loadTokenIndex loadBankState nextLoadTokenIndex nextLoadBankState
persistent requestBeatIndex receiveBeatIndex sumFlushCount
persistent pendingInvRms pendingTokenIndex pendingBankState
persistent readyTokenValid readyTokenIndex readyBankState
persistent outputActive outputTokenIndex outputBankState outputBeatIndex

if isempty(state)
    state = uint8(0);
    gammaMem = zeros(1, 1536, 'single');
    gammaWriteBeat = uint16(0);
    gammaLoaded = false;
    loadActive = false;
    loadTokenIndex = uint8(0);
    loadBankState = false;
    nextLoadTokenIndex = uint8(0);
    nextLoadBankState = false;
    requestBeatIndex = uint16(0);
    receiveBeatIndex = uint16(0);
    sumFlushCount = uint8(0);
    pendingInvRms = false;
    pendingTokenIndex = uint8(0);
    pendingBankState = false;
    readyTokenValid = false;
    readyTokenIndex = uint8(0);
    readyBankState = false;
    outputActive = false;
    outputTokenIndex = uint8(0);
    outputBankState = false;
    outputBeatIndex = uint16(0);
end

gBeat = zeros(1, 8, 'single');
gBeatValid = false;
captureInvRms = false;
captureInvRmsBank = false;
sramWriteAddr = uint16(0);
sramWriteBank = false;
sramWriteValid = false;
sramReadAddr = uint16(0);
sramReadBank = false;
sramReadValid = false;
ddrReadAddr = uint16(0);
ddrReadEn = false;
done = false;
busy = state ~= uint8(0);

if cfgGammaValid ~= 0
    gammaMem = iWriteBeat(gammaMem, cfgGammaBeat, gammaWriteBeat);
    if gammaWriteBeat == beatsPerToken - 1
        gammaWriteBeat = uint16(0);
        gammaLoaded = true;
    else
        gammaWriteBeat = gammaWriteBeat + uint16(1);
    end
end

if pendingBankState
    invRmsAccepted = invRmsAcceptedBank1;
else
    invRmsAccepted = invRmsAcceptedBank0;
end

switch state
    case uint8(0)
        if start ~= 0 && gammaLoaded
            state = uint8(1);
            loadActive = true;
            loadTokenIndex = uint8(0);
            loadBankState = false;
            nextLoadTokenIndex = uint8(1);
            nextLoadBankState = true;
            requestBeatIndex = uint16(0);
            receiveBeatIndex = uint16(0);
            sumFlushCount = uint8(0);
            pendingInvRms = false;
            readyTokenValid = false;
            outputActive = false;
            readyBankState = false;
            pendingBankState = false;
            outputBankState = false;
            outputBeatIndex = uint16(0);
            busy = true;
        end

    otherwise
        busy = true;

        if pendingInvRms && sumFlushCount > uint8(0)
            if sumFlushCount == uint8(1)
                captureInvRms = true;
                captureInvRmsBank = pendingBankState;
            end
            sumFlushCount = sumFlushCount - uint8(1);
        end

        if invRmsAccepted ~= 0 && pendingInvRms
            readyTokenValid = true;
            readyTokenIndex = pendingTokenIndex;
            readyBankState = pendingBankState;
            pendingInvRms = false;
        end

        if loadActive
            sramWriteBank = loadBankState;
            [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, loadTokenIndex, requestBeatIndex, beatsPerToken);
            if ddrDataValid ~= 0
                sramWriteAddr = receiveBeatIndex;
                sramWriteValid = true;

                if receiveBeatIndex == beatsPerToken - 1
                    if warmupFlushCycles == uint8(0)
                        captureInvRms = true;
                        captureInvRmsBank = loadBankState;
                    end
                    sumFlushCount = warmupFlushCycles;
                    pendingInvRms = true;
                    pendingTokenIndex = loadTokenIndex;
                    pendingBankState = loadBankState;
                    loadActive = false;
                    requestBeatIndex = uint16(0);
                    receiveBeatIndex = uint16(0);
                else
                    receiveBeatIndex = receiveBeatIndex + uint16(1);
                end
            end
        end

        if outputActive
            sramReadAddr = outputBeatIndex;
            sramReadBank = outputBankState;
            sramReadValid = true;
            gBeat = iReadBeat(gammaMem, outputBeatIndex);
            gBeatValid = true;

            if outputBeatIndex == beatsPerToken - 1
                if outputTokenIndex == numTokens - uint8(1)
                    done = true;
                    state = uint8(0);
                    loadActive = false;
                    pendingInvRms = false;
                    readyTokenValid = false;
                    outputActive = false;
                    nextLoadTokenIndex = uint8(0);
                    nextLoadBankState = false;
                    requestBeatIndex = uint16(0);
                    receiveBeatIndex = uint16(0);
                    outputBeatIndex = uint16(0);
                    sumFlushCount = uint8(0);
                    busy = false;
                else
                    outputActive = false;
                    outputBeatIndex = uint16(0);
                end
            else
                outputBeatIndex = outputBeatIndex + uint16(1);
            end
        end

        if ~outputActive && readyTokenValid
            outputActive = true;
            outputTokenIndex = readyTokenIndex;
            outputBankState = readyBankState;
            outputBeatIndex = uint16(0);
            readyTokenValid = false;
        end

        if state ~= uint8(0) && ~loadActive && nextLoadTokenIndex < numTokens
            canStartLoad = iCanStartLoad(nextLoadBankState, outputActive, outputBankState, pendingInvRms, pendingBankState, readyTokenValid, readyBankState);
            if canStartLoad
                loadActive = true;
                loadTokenIndex = nextLoadTokenIndex;
                loadBankState = nextLoadBankState;
                nextLoadTokenIndex = nextLoadTokenIndex + uint8(1);
                nextLoadBankState = ~nextLoadBankState;
                requestBeatIndex = uint16(0);
                receiveBeatIndex = uint16(0);
                [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, loadTokenIndex, requestBeatIndex, beatsPerToken);
            end
        end
end

end

function mem = iWriteBeat(mem, beat, beatIndex)
baseIndex = uint16(bitshift(beatIndex, 3));
for laneIndex = uint16(0):uint16(7)
    mem(baseIndex + laneIndex + uint16(1)) = beat(laneIndex + uint16(1));
end
end

function beat = iReadBeat(mem, beatIndex)
beat = zeros(1, 8, 'single');
baseIndex = uint16(bitshift(beatIndex, 3));
for laneIndex = uint16(0):uint16(7)
    beat(laneIndex + uint16(1)) = mem(baseIndex + laneIndex + uint16(1));
end
end

function [ddrReadEn, ddrReadAddr, requestBeatIndex] = iIssueRead(ddrReadEn, ddrReadAddr, tokenIndex, requestBeatIndex, beatsPerToken)
if requestBeatIndex < beatsPerToken
    ddrReadEn = true;
    ddrReadAddr = uint16(uint16(tokenIndex) * beatsPerToken + requestBeatIndex);
    requestBeatIndex = requestBeatIndex + uint16(1);
end
end

function canStartLoad = iCanStartLoad(targetBank, outputActive, outputBankState, pendingInvRms, pendingBankState, readyTokenValid, readyBankState)
occupied = false;

if outputActive && outputBankState == targetBank
    occupied = true;
end

if pendingInvRms && pendingBankState == targetBank
    occupied = true;
end

if readyTokenValid && readyBankState == targetBank
    occupied = true;
end

canStartLoad = ~occupied;
end
